INTEGRATE IN 10 LINES

Poll /api/live every 5 seconds to receive goal alerts the moment they are detected — before official confirmation arrives.

JavaScript 
// Receive goal alerts in real time
const BASE = 'https://kaching-tell-production.up.railway.app';

setInterval(async () => {
  const res  = await fetch(BASE + '/api/live');
  const data = await res.json();

  data.recent.forEach(detection => {
    if (detection.status === 'pending' || detection.status === 'verified') {
      console.log('GOAL — ' + detection.scoringTeam);
      console.log('Match: ' + detection.matchName);
      console.log('Score: ' + detection.scoreAtDetection);
      console.log('Clock: ' + detection.matchClockFormatted);
      console.log('Confidence: ' + detection.confidence);
      console.log('Lead time: ' + (detection.leadTimeMs
        ? Math.round(detection.leadTimeMs / 1000) + 's before official'
        : 'pending confirmation'));
      console.log('Proof: https://solscan.io/tx/' + detection.txSig);
    }
  });
}, 5000);
